country_event = {
	id = 100400
	title = "Основание торговой компании!"
	desc = " "
	picture = "chinese_concession"
	trigger = {
		tag = ENG
                OR = {
		     any_owned_province = {
			  is_core = CHI
		     }
                     any_owned_province = {
			  is_core = YNN
		     }
                     any_owned_province = {
			  is_core = GXI
		     }
                     any_owned_province = {
			  is_core = XBI
		     }
                }
		NOT = {
			is_our_vassal = ENC
			exists = ENC
		}
	}
	immediate = {
		any_owned = {
			limit = {
				OR = {
                                     is_core = CHI
                                     is_core = YNN
                                     is_core = GXI
                                     is_core = XBI
                                }
			}
			secede_province = ENC
			change_owner = ENC
		}
		set_country_flag = has_trade_company
	}
	option = {
		name = "Отлично"
                prestige = 100
                diplomatic_influence = {
                       who = ENC
                       value = 500
                }
                create_vassal = ENC
                ENC = {
                      government = colonial_company
                      tech_school = colonial_school
                }
	}
}

                     any_owned_province = {
			  is_core = XBI
		     }
                }
		OR = {
			is_our_vassal = ENC
			is_our_vassal = ASC
			is_our_vassal = GSC
			is_our_vassal = FRC
			is_our_vassal = SPC
			is_our_vassal = ITC
			is_our_vassal = RUC
			is_our_vassal = TCH
			is_our_vassal = USC
			is_our_vassal = JPC
			is_our_vassal = SCC
			is_our_vassal = BRC
                        is_our_vassal = NEC
			is_our_vassal = PRC
			is_our_vassal = MXC
		}
	}
	immediate = {
		any_country = {
			limit = {
				government = colonial_company
				vassal_of = THIS
                                NOT = {
                                      tag = BOC
                                      tag = FOC
                                }
			}
			country_event = 100402
		}
		set_country_flag = has_trade_company
	}
	option = {
		name = "Отлично"
	}
}
